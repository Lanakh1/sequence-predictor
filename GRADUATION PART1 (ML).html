    sequences = sequences_file.read().split('\n')
    sequences = list(filter(None, sequences))  # This removes empty sequences.

# Determine the maximum sequence length
max_sequence_length = max(len(seq) for seq in sequences)

# Initialize the LabelEncoder
integer_encoder = LabelEncoder()

# Encode sequences and store them
encoded_sequences = []

# Loop through your sequences
for sequence in sequences:
    # Pad the sequence to the maximum length with a placeholder character (e.g., 'N')
    padded_sequence = sequence.ljust(max_sequence_length, 'N')
    
    # Encode the sequence
    integer_encoded = integer_encoder.fit_transform(list(padded_sequence))
    encoded_sequences.append(integer_encoded)

# Stack the encoded sequences into a 2D NumPy array
encoded_sequences = np.array(encoded_sequences)

# Initialize the OneHotEncoder
one_hot_encoder = OneHotEncoder(categories='auto')

# One-hot encode all sequences at once
one_hot_encoded = one_hot_encoder.fit_transform(encoded_sequences.reshape(-1, 1))

# Reshape the one-hot encoded sequences into a 3D NumPy array
input_features = one_hot_encoded.toarray().reshape(len(sequences), max_sequence_length, -1)

def integrated_gradients(model, x, steps=50):
    x = tf.convert_to_tensor(x, dtype=tf.float32)

    # Define the baseline tensor as all zeros
    baseline = tf.zeros_like(x)

    # Create intermediate tensors along the path from baseline to input
    alphas = tf.linspace(start=0.0, stop=1.0, num=steps+1)
    interpolated_inputs = baseline + (x - baseline) * alphas[:, tf.newaxis, tf.newaxis]

    # Compute gradients of the model output with respect to the interpolated inputs
    interpolated_inputs = tf.Variable(interpolated_inputs)
    with tf.GradientTape() as tape:
        tape.watch(interpolated_inputs)
        logits = model(interpolated_inputs)
    gradients = tape.gradient(logits, interpolated_inputs)

    # Compute the integrated gradients
    integrated_gradients = tf.reduce_mean(gradients, axis=0) * (x - baseline)

    # Compute the magnitudes of the integrated gradients
    saliency_map = tf.reduce_sum(integrated_gradients, axis=-1)

    return saliency_map.numpy()

def predict_sequence(_):
    with output:
        output.clear_output()
        sequence = sequence_input.value

        padded_sequence = sequence.ljust(max_sequence_length, 'N')

        integer_encoded = [bases.index(base) for base in padded_sequence]

        # Initialize the OneHotEncoder with all possible categories
        one_hot_encoder = OneHotEncoder(categories=[range(len(bases))], sparse=False)

        # One-hot encode the sequence

        # Test the model on the unseen sequence

        # Get the predicted label (0 or 1)
